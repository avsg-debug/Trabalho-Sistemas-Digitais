Início

Este site resume o projeto Jogo da Velha com Arduino, que utiliza LEDs e botões para criar um tabuleiro físico, aplicando conceitos de lógica digital e programação embarcada.

Descrição do Projeto

Motivação: aprender eletrônica digital e lógica computacional de forma prática e interativa.

Objetivo: construir um tabuleiro de jogo da velha usando LEDs para as jogadas e botões como entrada dos dois jogadores.

Como funciona:

  • O jogador aperta um botão;
  • O Arduino identifica a posição;
  • O LED correspondente acende;
  • O sistema alterna entre jogadores;
  • O código verifica vitória ou empate.

Passo a passo

  1. Montar LEDs: matriz 3×3 com resistores ligados aos pinos digitais.
  2. Conectar botões: usar INPUT_PULLUP.
  3. Conectar ao Arduino: organizar a protoboard.
  4. Testar componentes: LEDs e botões.
  5. Carregar o código final.

Código (trecho principal)

int potenciometro = 0;
int ledAceso = 0;
int inicio = 1;
int tabuleiro[9];
int jogador = 1;
int i;
int botao = 11;
int pos = 0;

void setup() {
  for (i = 2; i < 11; i++) pinMode(i, OUTPUT);
  pinMode(botao, INPUT_PULLUP);

  for (i = 0; i < 9; i++) tabuleiro[i] = 0;
}

void desenhaTabuleiro() {
  for (i = 0; i < 9; i++) {
    int led = i + 2;

    if (tabuleiro[i] == 0) digitalWrite(led, LOW);
    else if (tabuleiro[i] == 1) digitalWrite(led, (millis() / 300) % 2);
    else if (tabuleiro[i] == 2) digitalWrite(led, HIGH);
  }
}

int verificaVitoria() {
  int v[8][3] = {
    {0,1,2},{3,4,5},{6,7,8},
    {0,3,6},{1,4,7},{2,5,8},
    {0,4,8},{2,4,6}
  };

  for (int k = 0; k < 8; k++) {
    int a = v[k][0], b = v[k][1], c = v[k][2];

    if (tabuleiro[a] != 0 && tabuleiro[a] == tabuleiro[b] && tabuleiro[b] == tabuleiro[c])
      return tabuleiro[a];
  }
  return 0;
}

void loop() {
  potenciometro = analogRead(A5);
  pos = map(potenciometro, 0, 1023, 0, 8);
  ledAceso = pos + 2;

  desenhaTabuleiro();

  if (tabuleiro[pos] == 0)
    digitalWrite(ledAceso, (millis() / 100) % 2);

  if (digitalRead(botao) == LOW && tabuleiro[pos] == 0) {
    delay(250);
    tabuleiro[pos] = jogador;
    jogador = (jogador == 1 ? 2 : 1);
  }

  int vencedor = verificaVitoria();
  if (vencedor != 0) {
    for (int n = 0; n < 6; n++) {
      for (int i = 2; i < 11; i++) digitalWrite(i, HIGH);
      delay(200);
      for (int i = 2; i < 11; i++) digitalWrite(i, LOW);
      delay(200);
    }

    for (int i = 0; i < 9; i++) tabuleiro[i] = 0;
    jogador = 1;
  }
}
